pipeline {
    agent any
    tools{
        maven 'maven3'
    }
    stages{
        stage('Build'){
            steps{
                sh 'mvn clean package'
            }
            post{
                success{
                    echo 'archieving war file'
                    archiveArtifacts artifacts: 'target/*.war'
                    echo 'success full'
                }
            }
        }      
    }
}
